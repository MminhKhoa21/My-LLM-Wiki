Circuit breaker ở mức code
class CircuitBreaker:
def call(self, fn, *args, **kwargs):
if self.state == "OPEN":
if not self.ready_to_probe():
raise CircuitOpenError()
self.state = "HALF_OPEN"
try:
result = fn(*args, **kwargs)
self.record_success()
return result
except Exception:
self.record_failure()
raise
